import requests def get_github_repos(username): url = f"https://api.github.com/users/{username}/repos" response = requests.get(url) if response.status_code == 200: repos = response.json() for repo in repos: print(f"Repo: {repo['name']} - {repo['html_url']}") else: print("Error fetching repositories. Check username and try again.") if __name__ == "__main__": username = input("Enter GitHub username: ") get_github_repos(username)
